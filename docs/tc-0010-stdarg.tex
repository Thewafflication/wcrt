% TC-0010 fragment; included by c89-test-report.tex.
\def\TCID{TC-0010}
\def\TCTitle{C89 variable arguments}
\def\TCPurpose{Verify the variable-argument declarations and ABI traversal of
promoted integers, floating values, pointers, and aggregates.}
\def\TCPriority{\TCPriorityHigh}
\def\TCRequirementRef{REQ-0010}
\def\TCDesignRef{ANSI X3J11 C89 draft 4.8 and 4.8.1.1 through 4.8.1.3.}
\def\TCFeatureRef{\texttt{include/stdarg.h}; \texttt{tests/c89/stdarg.c};
\texttt{tests/c89/run-tc-0010.ps1}.}
\def\TCPreconditions{TinyCC is available and supplies its ABI-aware variadic
builtins.}
\def\TCEnvironment{C89 presence compilation and behavioral execution against
WCRT and Microsoft UCRT.}
\def\TCAssumptions{Independent copying of active \texttt{va\_list} objects is
not required by C89 and is not tested.}
\def\TCInputData{Promoted char, short, and float values; long; pointer; pair
aggregate; repeated calls; and an empty tail.}
\def\TCInitialState{No variable-argument traversal is active.}
\def\TCProcedure{\par\begin{enumerate}
\item Compile the presence source as C89 without host header fallback.
\item Build and execute \texttt{tests/c89/stdarg.c} against both targets.
\item Verify values after every ABI transition and complete each traversal.
\end{enumerate}}
\def\TCExpectedResult{Every value is retrieved with its promoted type and both
targets complete repeated and empty-tail calls.}
\def\TCPostConditions{No variable-argument traversal remains active.}
\TCTemplate
